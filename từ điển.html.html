<input type="text" id="TimKiem">
<input type="button" value="tìm kiếm" onclick="HienThiTu()">
<p id="result"></p>
</body>
<script>
let Vn = ["táo","chuối","cam"]
let En = ["apple","banana","orange"]

function HienThiTu(){
    let TuDaNhap = document.getElementById("TimKiem").value.trim().toLowerCase();
    let e = ""
    for(let x=0; x < En.length; x++)
    {
        if (En[x] === TuDaNhap){
            e = En[x] + ":" + Vn[x]
            break
        }
        else{
          e = 'không tìm thấy'
        }
    }
    document.getElementById("result").innerHTML = e
}
</script>
